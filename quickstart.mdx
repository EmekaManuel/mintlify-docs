## Installation

Install the package:

<CodeGroup>
```bash npm
npm install sentinelx404
```

```bash yarn
yarn add sentinelx404
```

```bash bun
bun add sentinelx404
```
</CodeGroup>

## Next.js Setup

If you're using Next.js, add this to your `next.config.js`:

```javascript
/** @type {import('next').NextConfig} */
const nextConfig = {
  transpilePackages: ["sentinelx404"],
};

export default nextConfig;
```

<Warning>
Make sure to restart your Next.js dev server after adding this configuration.
</Warning>

## Basic Usage

Here's a complete example using the Blacklist feature:

```typescript
"use client";

import { X404Blacklist } from "sentinelx404/blacklist";
import { getGeolocationData } from "sentinelx404/utils";

async function authenticate() {
  // Get user's location (optional)
  const location = await getGeolocationData();

  // Authenticate - wallet is optional, modal will show automatically
  const result = await X404Blacklist({
    excluded_mints: ["scam_token_address"],
    max_holdings: {},
    geo_code: "false",
    geo_code_locs: "",
    coords: {
      latitude: location.latitude,
      longitude: location.longitude,
    },
  });

  if (result.success) {
    console.log("✅ Authenticated! Token:", result.token);
    // Token is automatically stored in localStorage
  } else {
    console.error("❌ Error:", result.error, result.message);
  }
}
```

## React Component Example

```typescript
"use client";

import { useState } from "react";
import { X404Blacklist } from "sentinelx404/blacklist";
import { getGeolocationData } from "sentinelx404/utils";
import type { X404AuthResult } from "sentinelx404";

export function AuthButton() {
  const [loading, setLoading] = useState(false);
  const [result, setResult] = useState<X404AuthResult | null>(null);

  const handleAuth = async () => {
    setLoading(true);
    try {
      const location = await getGeolocationData();
      
      const authResult = await X404Blacklist({
        excluded_mints: ["scam_token_address"],
        max_holdings: {},
        geo_code: "false",
        geo_code_locs: "",
        coords: {
          latitude: location.latitude,
          longitude: location.longitude,
        },
      });

      setResult(authResult);
    } catch (error) {
      console.error("Error:", error);
    } finally {
      setLoading(false);
    }
  };

  return (
    <div>
      <button onClick={handleAuth} disabled={loading}>
        {loading ? "Authenticating..." : "Authenticate"}
      </button>
      {result && (
        <div>
          {result.success ? (
            <p>✅ Authenticated! Token: {result.token}</p>
          ) : (
            <p>❌ Error: {result.error}</p>
          )}
        </div>
      )}
    </div>
  );
}
```

## Available Features

You can import any of the 7 features:

```typescript
import { X404Blacklist } from "sentinelx404/blacklist";
import { X404TimeLock } from "sentinelx404/timelock";
import { X404MultiToken } from "sentinelx404/multitoken";
import { X404Activity } from "sentinelx404/activity";
import { X404Tier } from "sentinelx404/tier";
import { X404NoDebt } from "sentinelx404/nodebt";
import { X404Age } from "sentinelx404/age";
```

Or import all at once:

```typescript
import {
  X404Blacklist,
  X404TimeLock,
  X404MultiToken,
  X404Activity,
  X404Tier,
  X404NoDebt,
  X404Age,
} from "sentinelx404";
```
